79static std::string CharToString(unsigned char c) {
80 using namespace std;
81 stringstream ss;
82 ss << setfill('0') << hex << setw(2) << (c & 0x00ff);
83 return ss.str();

90std::string Nmea2000Msg::to_string() const {
91 // Arrange to "pretty-print" the N2K message payload
92
93 // extract PGN from payload
94 uint64_t pgn = 0;
95 unsigned char* c = (unsigned char*)&pgn;
96 *c++ = payload.at(3);
97 *c++ = payload.at(4);
98 *c++ = payload.at(5);
99
100 std::string st;
101 size_t data_start = 12;
102 if (payload.at(0) == 0x94) data_start = 7;
103 size_t i = 0;
104 while (i < payload.size() - 1) {
105 if (i > data_start) {
106 for (int j = 0; j < 8; j++) {
107 st.append(CharToString(payload[i]));
108 st.append(" ");
109 i++;
110 if (i >= payload.size() - 1) {
111 break;
112 }
113 }
114 st.append(" ");
115 } else
116 i++;
117 }
118
119 std::stringstream ss1;
120 std::string spgn = " ";
121 ss1 << "n2000 pgn: " << pgn << " [ " << st << " ]";
122 return ss1.str();
123}

139std::string PluginMsg::to_string() const {
141 if (name == "position-fix") return name + ":" + ocpn::rtrim(message);
142
143 std::stringstream ss;
144 for (char c : ocpn::rtrim(message)) {
145 if (c >= ' ' && c <= '~')
146 ss << c;
147 else
148 ss << CharToString(c);
149 }
150 return name + ": " + ss.str();
151}
