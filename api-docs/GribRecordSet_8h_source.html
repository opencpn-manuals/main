162 GribRecordSet(unsigned int id) : m_Reference_Time(-1), m_ID(id) {
163 for (int i = 0; i < Idx_COUNT; i++) {
164 m_GribRecordPtrArray[i] = 0;
165 m_GribRecordUnref[i] = false;
166 }
167 }
168
169 virtual ~GribRecordSet() { RemoveGribRecords(); }
170
180 void SetUnRefGribRecord(int i, GribRecord *pGR) {
181 assert(i >= 0 && i < Idx_COUNT);
182 if (m_GribRecordUnref[i] == true) {
183 delete m_GribRecordPtrArray[i];
184 }
185 m_GribRecordPtrArray[i] = pGR;
186 m_GribRecordUnref[i] = true;
187 }
188
194 void RemoveGribRecords() {
195 for (int i = 0; i < Idx_COUNT; i++) {
196 if (m_GribRecordUnref[i] == true) {
197 delete m_GribRecordPtrArray[i];
198 }
199 }
200 }

204 time_t m_Reference_Time;
206 unsigned int m_ID;

216 GribRecord *m_GribRecordPtrArray[Idx_COUNT];
217
218private:

221 bool m_GribRecordUnref[Idx_COUNT];
