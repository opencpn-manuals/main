168void GoToPositionDialog::OnGoToPosOkClick(wxCommandEvent& event) {
169 double lat, lon;
170
171 if (m_MarkLatCtl->GetValue().Length() == 0) goto noGo;
172 if (m_MarkLonCtl->GetValue().Length() == 0) goto noGo;
173
174 lat = fromDMM(m_MarkLatCtl->GetValue());
175 lon = fromDMM(m_MarkLonCtl->GetValue());
176
177 if (lat == 0.0 && lon == 0.0) goto noGo;
178 if (lat > 80.0 || lat < -80.0) goto noGo;
179 if (lon > 180.0 || lon < -180.0) goto noGo;
180
181 if (m_hostCanvas)
182 gFrame->JumpToPosition(m_hostCanvas, lat, lon, m_hostCanvas->GetVPScale());
183 Hide();
184 event.Skip();
185 return;
186
187noGo:
188 wxBell();
189 event.Skip();
190 return;
191}

193void GoToPositionDialog::CheckPasteBufferForPosition() {
194 if (wxTheClipboard->Open()) {
195 wxTextDataObject data;
196 wxTheClipboard->GetData(data);
197 wxString pasteBuf = data.GetText();
198
199 PositionParser pparse(pasteBuf);
200
201 if (pparse.IsOk()) {
202 m_MarkLatCtl->SetValue(pparse.GetLatitudeString());
203 m_MarkLonCtl->SetValue(pparse.GetLongitudeString());
204 }
205 wxTheClipboard->Close();
206 }
207}
