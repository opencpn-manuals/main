171void GoToPositionDialog::OnGoToPosOkClick(wxCommandEvent& event) {
172 double lat, lon;
173
174 if (m_MarkLatCtl->GetValue().Length() == 0) goto noGo;
175 if (m_MarkLonCtl->GetValue().Length() == 0) goto noGo;
176
177 lat = fromDMM(m_MarkLatCtl->GetValue());
178 lon = fromDMM(m_MarkLonCtl->GetValue());
179
180 if (lat == 0.0 && lon == 0.0) goto noGo;
181 if (lat > 80.0 || lat < -80.0) goto noGo;
182 if (lon > 180.0 || lon < -180.0) goto noGo;
183
184 if (m_hostCanvas) {
185 top_frame::Get()->JumpToPosition(m_hostCanvas, lat, lon,
186 m_hostCanvas->GetVPScale());
187 }
188 Hide();
189 event.Skip();
190 return;
191
192noGo:
193 wxBell();
194 event.Skip();
195 return;
196}

198void GoToPositionDialog::CheckPasteBufferForPosition() {
199 if (wxTheClipboard->Open()) {
200 wxTextDataObject data;
201 wxTheClipboard->GetData(data);
202 wxString pasteBuf = data.GetText();
203
204 PositionParser pparse(pasteBuf);
205
206 if (pparse.IsOk()) {
207 m_MarkLatCtl->SetValue(pparse.GetLatitudeString());
208 m_MarkLonCtl->SetValue(pparse.GetLongitudeString());
209 }
210 wxTheClipboard->Close();
211 }
